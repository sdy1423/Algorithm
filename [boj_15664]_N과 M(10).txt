#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
vector<vector<int>> d;
int num[10];
int a[10];
void go(int index,int start, int n, int m) {
	if (index == m) {
		vector<int> temp;
		for (int i = 0; i < m; i++) {
			temp.push_back(num[a[i]]);
		}
		d.push_back(temp);
		return;
	}
	for (int i = start; i < n; i++) {
		a[index] = i;
		go(index + 1,i+1, n, m);
	}
}
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> num[i];
	}
	sort(num, num + n);
	go(0, 0, n, m);
	sort(d.begin(), d.end());
	d.erase(unique(d.begin(), d.end()), d.end());
	for (auto& k : d) {
		for (int i = 0; i < m; i++) {
			cout << k[i];
			if (i != m - 1) cout << ' ';
		}
		cout << '\n';
	}
	return 0;
}
